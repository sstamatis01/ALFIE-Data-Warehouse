# ==========================================
# Logfire telemetry (optional)
# ==========================================
# When set, DW sends traces/logs to Logfire so partners can correlate with Task Manager, orchestrator, UI.
# Get a write token from https://logfire.pydantic.dev (Project Settings > Write Tokens).
# Leave unset to run without telemetry.
# LOGFIRE_TOKEN=your-write-token

# ==========================================
# GDPR delete endpoint (optional but recommended)
# ==========================================
# If set, `DELETE /gdpr/users/{user_id}` requires header:
#   X-GDPR-Token: <GDPR_DELETE_TOKEN>
# GDPR_DELETE_TOKEN=change-me

# ==========================================
# Keycloak (optional) - used by GDPR delete to remove the user in Keycloak too
# ==========================================
# If KEYCLOAK_URL + KEYCLOAK_REALM + KEYCLOAK_ADMIN_CLIENT_ID are set, the DW will attempt to delete
# the Keycloak user when you call `DELETE /gdpr/users/{user_id}`.
#
# IMPORTANT: current implementation assumes {user_id} is the Keycloak user UUID (the OIDC `sub`).
#
# Base URL of Keycloak, without trailing slash (examples):
# KEYCLOAK_URL=http://localhost:8080
# KEYCLOAK_URL=https://keycloak.my-domain.tld
#
# Realm name where the users live:
# KEYCLOAK_REALM=my-realm
#
# Admin credentials option A (recommended): client credentials
# The client must be allowed to call the Admin REST API and have realm-management roles (typically `manage-users`).
# KEYCLOAK_ADMIN_CLIENT_ID=admin-cli-or-service-client
# KEYCLOAK_ADMIN_CLIENT_SECRET=your-client-secret
#
# Admin credentials option B: password grant (works but less recommended for service-to-service)
# KEYCLOAK_ADMIN_CLIENT_ID=admin-cli
# KEYCLOAK_ADMIN_USERNAME=admin
# KEYCLOAK_ADMIN_PASSWORD=admin-password
